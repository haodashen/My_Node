收货日期：<input type="text" id="purchaseDate" name="purchaseDate" 
	required=true size="18" class="easyui-datebox" data-options="editable:false" />
	
$("#purchaseDate").datebox("setValue",genTodayStr()); // 设置当前日期

function genTodayStr(){
	var date = new Date();
	var dateStr = date.getFullYear()+"-"+formatZero(date.getMonth()+1)+"-"+formatZero(date.getDate());
	return dateStr;
}
function formatZero(n){
	if(n<10){
		return "0"+n;
	}else{
		return n;
	}
}
